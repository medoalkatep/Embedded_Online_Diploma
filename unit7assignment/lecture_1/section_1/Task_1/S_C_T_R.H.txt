#ifndef UTILS_H_
#define UTILS_H_

#define set_bit(reg , bit)  (reg |= (1 << bit))

#define clear_bit(reg , bit)  (reg &= ~(1 << bit))

#define read_bit(reg , bit)  ( (reg >> bit) & 1 )

#define toggle_bit(reg , bit)  (reg ^= (1 << bit))




#endif /* UTILS_H_ */
