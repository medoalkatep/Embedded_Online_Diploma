#include "MemMap_T1.h"
#include "util/delay.h"

int main(void)
{
    DDRA = 0b11111111 ;
	
    while (1) 
    {
		PORTA |= (1 << 0) ;
		_delay_ms(1000) ;
		PORTA |= (1 << 1) ;
		_delay_ms(1000) ;
		PORTA |= (1 << 2) ;
		_delay_ms(1000) ;
		PORTA |= (1 << 3) ;
		_delay_ms(1000) ;
		PORTA |= (1 << 4) ;
		_delay_ms(1000) ;
		PORTA |= (1 << 5) ;
		_delay_ms(1000) ;
		PORTA |= (1 << 6) ;
		_delay_ms(1000) ;
		PORTA |= (1 << 7) ;
		_delay_ms(1000) ;
		
		PORTA &=~(1 << 7 ) ;
		_delay_ms(1000) ;
		PORTA &=~(1 << 6 ) ;
		_delay_ms(1000) ;
		PORTA &=~(1 << 5 ) ;
		_delay_ms(1000) ;
		PORTA &=~(1 << 4 ) ;
		_delay_ms(1000) ;
		PORTA &=~(1 << 3 ) ;
		_delay_ms(1000) ;
		PORTA &=~(1 << 2 ) ;
		_delay_ms(1000) ;
		PORTA &=~(1 << 1 ) ;
		_delay_ms(1000) ;
		PORTA &=~(1 << 0 ) ;
		_delay_ms(1000) ;
		
		
		  
    }
}
